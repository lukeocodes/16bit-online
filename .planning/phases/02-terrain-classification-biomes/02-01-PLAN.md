```typescript
import { getWorldMap } from "./queries.js";
import { BiomeType } from "./types.js";
import { CHUNK_SIZE } from "./constants.js";

// Biomes that block all movement (player and NPC)
export const BLOCKING_BIOMES = new Set<number>([
  BiomeType.DEEP_OCEAN,
  BiomeType.SHALLOW_OCEAN,
  BiomeType.SNOW_PEAK,
  BiomeType.RIVER,
  BiomeType.LAKE,
]);

// 7 elevation bands quantizing the 0.0-1.0 continuous elevation range
export const ELEVATION_BANDS = [
  { min: 0.00, max: 0.15, level: 0, name: "deep_water" },
  { min: 0.15, max: 0.30, level: 1, name: "shallow_water" },
  { min: 0.30, max: 0.45, level: 2, name: "lowland" },
  { min: 0.45, max: 0.60, level: 3, name: "plains" },
  { min: 0.60, max: 0.75, level: 4, name: "highland" },
  { min: 0.75, max: 0.90, level: 5, name: "mountain" },
  { min: 0.90, max: 1.00, level: 6, name: "peak" },
];

export { ELEVATION_STEP_HEIGHT } from "./constants.js";

/**
 * Convert continuous elevation (0.0-1.0) to discrete band level (0-6).
 */
export function getElevationBand(elevation: number): number {
  for (const band of ELEVATION_BANDS) {
    if (elevation < band.max) return band.level;
  }
  return ELEVATION_BANDS[ELEVATION_BANDS.length - 1].level;
}

/**
 * Check if a tile position is walkable.
 * Converts tile coordinates to chunk coordinates for biomeMap lookup.
 * Returns false for out-of-bounds or blocking biome types.
 */
export function isWalkable(tileX: number, tileZ: number): boolean {
  const world = getWorldMap();
  if (!world) return false;

  const chunkX = Math.floor(tileX / CHUNK_SIZE);
  const chunkZ = Math.floor(tileZ / CHUNK_SIZE);

  if (chunkX < 0 || chunkX >= world.width || chunkZ < 0 || chunkZ >= world.height) {
    return false;
  }

  const biome = world.biomeMap[chunkZ * world.width + chunkX];
  return !BLOCKING_BIOMES.has(biome);
}

/**
 * Check if a biome type is walkable (for use without world map context).
 */
export function isBiomeWalkable(biome: number): boolean {
  return !BLOCKING_BIOMES.has(biome);
}
```

**Step 1: Create rivers.ts**

Create `packages/server/src/world/rivers.ts` implementing the full river/lake algorithm.

The function signature:
```typescript
import alea from "alea";
import { BiomeType, LandType } from "./types.js";
import {
  RIVER_SOURCE_ELEVATION_MIN,
  RIVER_SOURCE_MOISTURE_MIN,
  RIVER_MAX_STEPS,
  RIVER_WIDTH_FACTOR,
  LAKE_MIN_SIZE,
  LAKE_MAX_FILL_DEPTH,
  NUM_RIVER_SOURCES,
} from "./constants.js";

/**
 * Generate rivers and lakes on the world map.
 * Modifies biomeMap in place, adding RIVER and LAKE biome values.
 * Also sets RIVER_VALLEY on walkable land chunks adjacent to rivers.
 *
 * Algorithm:
 * 1. Detect lake basins (local elevation minima) and flood-fill
 * 2. Select river source candidates (high elevation + high moisture)
 * 3. Trace each river downhill to ocean or lake
 * 4. Mark river-adjacent land chunks as RIVER_VALLEY
 */
export function generateRiversAndLakes(
  seed: number,
  biomeMap: Uint8Array,
  elevation: Float32Array,
  moisture: Float32Array,
  landmask: Uint8Array,
  width: number,
  height: number,
): void
```

**Lake detection algorithm:**
1. Scan all land chunks. A chunk is a basin candidate if all 4 cardinal neighbors have elevation >= its elevation.
2. From each basin, flood-fill outward: add neighbors whose elevation is <= (basin_elevation + LAKE_MAX_FILL_DEPTH).
3. If flooded area >= LAKE_MIN_SIZE, mark all flooded chunks as BiomeType.LAKE in biomeMap.
4. Use alea(`${seed}-lakes`) for any random decisions. Deterministic from seed.

**River tracing algorithm:**
1. Create a seeded PRNG with alea(`${seed}-rivers`).
2. Collect all land chunks where elevation > RIVER_SOURCE_ELEVATION_MIN and moisture > RIVER_SOURCE_MOISTURE_MIN.
3. Shuffle candidates deterministically using the PRNG.
4. Take first NUM_RIVER_SOURCES candidates.
5. For each source, trace downhill:
   a. Create a visited Set to track visited chunk indices.
   b. Start at source chunk. flowAccumulation = 1.
   c. Each step: look at 4 cardinal neighbors (dx,dz in [(-1,0),(1,0),(0,-1),(0,1)]).
   d. Pick the neighbor with the lowest elevation that has not been visited.
   e. If no lower unvisited neighbor exists, terminate (stuck in basin -- lake should have caught it).
   f. Mark current chunk as BiomeType.RIVER in biomeMap.
   g. Track flow direction (dx, dz from prev to current) for perpendicular width expansion.
   h. Width = Math.min(8, Math.max(1, Math.floor(flowAccumulation / RIVER_WIDTH_FACTOR))).
   i. If width > 1, mark chunks perpendicular to flow direction. For flow direction (dx, 0), expand in z direction: mark chunks at (cx, cz-1) through (cx, cz-halfWidth) and (cx, cz+1) through (cx, cz+halfWidth). For flow direction (0, dz), expand in x direction similarly. halfWidth = Math.floor(width / 2).
   j. Only mark expansion chunks as RIVER if they are within bounds and are land (landmask >= LAND).
   k. Increment flowAccumulation.
   l. Move to next chunk. Add to visited set.
   m. Terminate if: out of bounds, reached ocean (landmask[idx] < LandType.LAND), reached a LAKE chunk, or steps > RIVER_MAX_STEPS, or already visited.

**River valley post-processing:**
After all rivers are traced, scan the biomeMap. For every chunk that is BiomeType.RIVER, check its 8 neighbors (cardinal + diagonal). If a neighbor is land and NOT already RIVER or LAKE, set it to BiomeType.RIVER_VALLEY. This overwrites the previous biome (e.g., TEMPERATE_FOREST becomes RIVER_VALLEY for the strip next to the river).
