```typescript
// LandType enum (stored in Uint8Array landmask)
export const enum LandType {
  DEEP_OCEAN = 0,

  LAKE = 3,

// BiomeType enum (stored in Uint8Array biomeMap)
export const enum BiomeType {
  DEEP_OCEAN = 0,
  SHALLOW_OCEAN = 1,
  BEACH = 2,
  TEMPERATE_GRASSLAND = 3,
  TEMPERATE_FOREST = 4,
  DENSE_FOREST = 5,
  BOREAL_FOREST = 6,
  MOUNTAIN = 7,
  SNOW_PEAK = 8,
  TUNDRA = 9,
  DESERT = 10,
  SCRUBLAND = 11,
  SWAMP = 12,
  HIGHLAND = 13,
  MEADOW = 14,
  RIVER_VALLEY = 15,
}

// POI type taxonomy (Claude's discretion from CONTEXT.md)
export type POIType = "ruin" | "cave" | "landmark" | "resource" | "settlement" | "strait";

export interface POI {
  type: POIType;
  x: number;       // chunk coordinate
  z: number;       // chunk coordinate
  subtype?: string;
  connects?: [number, number]; // for straits: [regionIdA, regionIdB]
}

export type RaceType = "human" | "elf" | "dwarf";

export interface ContinentDef {
  id: string;
  name: string;
  race: RaceType;
  centerX: number;  // chunk coordinate (array-index space, 0-based)
  centerZ: number;  // chunk coordinate (array-index space, 0-based)
  radius: number;   // base radius in chunks
}

export interface Continent {
  id: string;
  name: string;
  race: RaceType;
  centerX: number;
  centerZ: number;
  radius: number;
  chunkCount: number;  // total land chunks in this continent
}

export interface Region {
  id: number;
  name: string;
  continentId: string;   // 'elf', 'dwarf', 'human', 'ocean'
  centerX: number;       // chunk coordinate
  centerZ: number;       // chunk coordinate
  biome: BiomeType;
  isLand: boolean;
  pois: POI[];
  chunkCount: number;    // computed after region assignment
}

export interface WorldConfig {
  seed: number;
  width: number;   // in chunks
  height: number;  // in chunks
}

export interface WorldMap {
  seed: number;
  width: number;
  height: number;
  continents: Continent[];
  regions: Region[];
  landmask: Uint8Array;       // indexed by [z * width + x], values are LandType
  elevation: Float32Array;    // indexed by [z * width + x], range 0.0-1.0
  moisture: Float32Array;     // indexed by [z * width + x], range 0.0-1.0
  temperature: Float32Array;  // indexed by [z * width + x], range 0.0-1.0
  regionMap: Uint16Array;     // indexed by [z * width + x], values are region IDs
  continentMap: Uint8Array;   // indexed by [z * width + x], values: 0=ocean, 1-3=continent index
  biomeMap: Uint8Array;       // indexed by [z * width + x], values are BiomeType
}
```

Use `const enum` for LandType and BiomeType for zero-cost abstraction at runtime. Export everything as named exports. Follow codebase conventions: PascalCase for interfaces, camelCase for properties.

**Step 2: Implement `packages/server/src/world/continents.ts` (GREEN phase).**

Implement these exported functions:

```typescript
import { createNoise2D } from "simplex-noise";
import alea from "alea";
```

1. **`fbm(noise2D, x, y, octaves, lacunarity, gain)`** -- fractal Brownian motion utility. Returns normalized value in [-1, 1]. Use the exact implementation from 01-RESEARCH.md Pattern 2.

2. **`placeContinents(width, height)`** -- returns `ContinentDef[]` with 3 continents in triangular layout:
   - Elf: top center (centerX = width/2, centerZ = height/2 - CONTINENT_OFFSET)
   - Dwarf: bottom-left (centerX = width/2 - CONTINENT_OFFSET * cos(PI/6), centerZ = height/2 + CONTINENT_OFFSET * sin(PI/6))
   - Human: bottom-right (centerX = width/2 + CONTINENT_OFFSET * cos(PI/6), centerZ = height/2 + CONTINENT_OFFSET * sin(PI/6))
   - Names: "Faerwood" (elf), "Khazrath" (dwarf), "Aethermere" (human)
   - All with radius = CONTINENT_RADIUS (175)

3. **`generateContinents(seed, config)`** -- returns `{ landmask, continentMap, continentDefs }`:
   - Create noise functions with unique seeds: `alea(\`${seed}-land\`)`, `alea(\`${seed}-coast\`)`
   - For each chunk (cx, cz), compute land value from radial gradient + fBm noise + coastal detail
   - Also generate island clusters between each continent pair: for each pair, place ISLANDS_PER_PAIR small radial gradients (radius ISLAND_CLUSTER_RADIUS) at midpoints offset by noise
   - Classify: DEEP_OCEAN (< SHALLOW_THRESHOLD), SHALLOW_OCEAN (< LAND_THRESHOLD), LAND (>= LAND_THRESHOLD)
   - Build continentMap: for each land chunk, assign to nearest continent (1-indexed); ocean = 0

4. **`generateElevation(seed, config, landmask)`** -- returns `Float32Array`:
   - Create noise with `alea(\`${seed}-elevation\`)`
   - For each chunk, compute fBm value, normalize to [0, 1]
   - Land chunks get boosted elevation (add 0.3, clamp to 1.0)
   - Ocean chunks get reduced elevation (multiply by 0.3)

5. **`generateMoisture(seed, config, landmask)`** -- returns `Float32Array`:
   - Create noise with `alea(\`${seed}-moisture\`)`
   - Similar to elevation but uses MOISTURE_NOISE_SCALE
   - Coastal chunks (land adjacent to ocean) get +0.1 moisture boost

6. **`generateTemperature(seed, config, landmask)`** -- returns `Float32Array`:
   - Create noise with `alea(\`${seed}-temperature\`)`
   - Latitude influence: chunks near the vertical center of the world are warmer, edges are cooler
   - Formula: `base = 1.0 - Math.abs(cz - height/2) / (height/2) * TEMPERATURE_LATITUDE_WEIGHT`
   - Add noise variation on top

**CRITICAL: Every `createNoise2D()` call MUST receive its own `alea()` instance with a unique seed string. NEVER use Math.random(). NEVER reuse a PRNG instance across noise functions.**
