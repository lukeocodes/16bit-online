<objective>
Wire the world generation into server startup and expose a query API for runtime chunk lookups.

Purpose: This plan completes TECH-01 and TECH-02 by making the world map accessible at runtime. The world map generates once at server boot (before the game loop starts) and exposes O(1) query functions that any server module (combat, spawning, chunk streaming) can use to determine what region, continent, and biome a chunk belongs to. This is the integration layer between the pure generation pipeline and the live game server.

```typescript
import { generateWorld } from "./worldgen.js";
import type { WorldMap, Region, Continent, BiomeType, WorldConfig } from "./types.js";

let worldMap: WorldMap | null = null;

/**
 * Initialize the world map from a seed. Call once at server startup.
 * Generates the complete world synchronously (~100-500ms).
 */
export function initWorldMap(seed: number, config?: Partial<WorldConfig>): void {
  const start = performance.now();
  worldMap = generateWorld(seed, config);
  const elapsed = (performance.now() - start).toFixed(0);
  console.log(`[World] Map initialized in ${elapsed}ms: ` +
    `${worldMap.continents.length} continents, ` +
    `${worldMap.regions.length} regions`);
}

/** Get the full world map. Returns null if not initialized. */
export function getWorldMap(): WorldMap | null {
  return worldMap;
}

/** Get the region containing the given chunk. Returns null if out of bounds. */
export function getRegionForChunk(cx: number, cz: number): Region | null {
  if (!worldMap) return null;
  if (cx < 0 || cx >= worldMap.width || cz < 0 || cz >= worldMap.height) return null;
  const regionId = worldMap.regionMap[cz * worldMap.width + cx];
  return worldMap.regions[regionId] ?? null;
}

/** Get the continent containing the given chunk. Returns null if ocean or out of bounds. */
export function getContinentForChunk(cx: number, cz: number): Continent | null {
  if (!worldMap) return null;
  if (cx < 0 || cx >= worldMap.width || cz < 0 || cz >= worldMap.height) return null;
  const continentIdx = worldMap.continentMap[cz * worldMap.width + cx];
  if (continentIdx === 0) return null; // ocean
  return worldMap.continents[continentIdx - 1] ?? null;
}

/** Get the biome type for the given chunk. Returns DEEP_OCEAN if out of bounds. */
export function getBiomeForChunk(cx: number, cz: number): number {
  if (!worldMap) return 0; // DEEP_OCEAN
  if (cx < 0 || cx >= worldMap.width || cz < 0 || cz >= worldMap.height) return 0;
  return worldMap.biomeMap[cz * worldMap.width + cx];
}

/** Get a region by its ID. Returns null if invalid. */
export function getRegionById(id: number): Region | null {
  if (!worldMap) return null;
  return worldMap.regions[id] ?? null;
}
```

All functions use array index lookups on typed arrays -- guaranteed O(1).

```typescript
import { buildApp } from "./app.js";
import { config } from "./config.js";
import { connectRedis, disconnectRedis } from "./db/redis.js";
import { spawnInitialNpcs, cleanup as cleanupNpcs } from "./game/npcs.js";
import { startGameLoop, stopGameLoop } from "./game/world.js";
import { initWorldMap } from "./world/queries.js";

async function main() {
  const app = await buildApp();

  await connectRedis();

  // Generate world map from seed (deterministic, ~100-500ms)
  initWorldMap(config.world.seed);

  spawnInitialNpcs();
  startGameLoop();

  await app.listen({ host: config.server.host, port: config.server.port });

  const shutdown = async () => {
    stopGameLoop();
    cleanupNpcs();
    await disconnectRedis();
    await app.close();
    process.exit(0);
  };

  process.on("SIGINT", shutdown);
  process.on("SIGTERM", shutdown);
}

main().catch((err) => {
  console.error("Failed to start server:", err);
  process.exit(1);
});
```
